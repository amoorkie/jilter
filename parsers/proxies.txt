# Формат: host:port:username:password (опционально)
# Или просто: host:port

# Примеры (замените на реальные прокси):
# 192.168.1.100:8080
# 10.0.0.1:3128:user:pass
# proxy1.example.com:8080
# proxy2.example.com:3128:username:password
